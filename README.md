```bash
# 克隆源码到本地
git@github.com:InvisibleFuture/satori.git

# 进入项目目录
cd satori

# 使用 npm 安装依赖包
npm install

# 也许你应该升级一下依赖包
npm update

# 编译
npm run build

# 运行服务端渲染: http://127.0.0.1:3000
npm run preview

# 如果你需要本地开发
npm run dev

# docs
# vue3: https://vuejs.org
# Nuxt3: https://v3.nuxtjs.org
# Windcss: https://windicss.org
```
